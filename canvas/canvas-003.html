たくさんの矩形（正方形）を発生させる


					
                        const canvas = document.getElementById("canvas");
                        var ctx = canvas.getContext("2d");
                        
                        const w = 640;
                        const h = 335;
                        
						for (let i = 0; i < 300; i++) {
							var rnd_x=Math.floor(Math.random() * w); 
							var rnd_y=Math.floor(Math.random() * h);
							ctx.fillRect(rnd_x, rnd_y, 10, 10); 
						}
